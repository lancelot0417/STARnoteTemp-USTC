    \clearpage

      \starnote@section*{#3}%

      \IfNoValueTF{#2}{\starnote@section{#3}}{\starnote@section[#2]{#3}}%

      \starnote@subsection*{#3}%
    }{%

    }%
  }%

    \IfBooleanTF{#1}{%
      \starnote@subsubsection*{#3}%

    }%
  }%
